import json
import chromadb

print("Loading data...")
with open("data/_processed/all_chunks.json", "r", encoding="utf-8") as f:
    data = json.load(f)

chunks = data["chunks"]
print(f"Loaded {len(chunks)} chunks")

# Dedupe by ID
seen = set()
unique_chunks = []
for c in chunks:
    if c["id"] not in seen:
        seen.add(c["id"])
        unique_chunks.append(c)
print(f"After dedup: {len(unique_chunks)} chunks ({len(chunks) - len(unique_chunks)} duplicates removed)")

print("Creating ChromaDB collection...")
client = chromadb.PersistentClient(path="./chroma_db")
collection = client.get_or_create_collection(
    name="pm_tools",
    metadata={"hnsw:space": "cosine"}
)

def clean_metadata(meta):
    cleaned = {}
    for k, v in meta.items():
        if isinstance(v, (str, int, float, bool)):
            cleaned[k] = v
        elif isinstance(v, list):
            cleaned[k] = ", ".join(str(i) for i in v)
        elif v is None:
            continue
    return cleaned

BATCH = 5000
for i in range(0, len(unique_chunks), BATCH):
    batch = unique_chunks[i:i+BATCH]
    collection.add(
        ids=[c["id"] for c in batch],
        documents=[c["text"] for c in batch],
        metadatas=[clean_metadata(c["metadata"]) for c in batch]
    )
    print(f"Ingested {min(i + BATCH, len(unique_chunks))}/{len(unique_chunks)}")

print(f"\nDone. Collection has {collection.count()} chunks.")

print("\n--- Test Query ---")
results = collection.query(
    query_texts=["Why did people switch from Jira to Linear?"],
    n_results=3
)
for doc, meta, dist in zip(results["documents"][0], results["metadatas"][0], results["distances"][0]):
    print(f"\n[{meta.get('app', '?')} | {meta.get('source', '?')} | dist: {round(dist, 3)}]")
    print(doc[:200])
